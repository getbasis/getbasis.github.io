// Bounce vertical

// @param   length             px        bounce size

//

    _bounce-vertical(4px, .5s);
  }
}

_extend-underline

//
// Extend the underline from the center.
// This mixin use the pseudo-elements.
//
.u-extend-underline {
  position: relative;

  &::after {
    _extend-underline();
  }

  +_hover() {
    &::after {
      width: 100%;
    }
  }
}
