_row

//
// The grid component
//
// Grid wrapper
//
_row();

//
// Setting column width
//
// @param  int  columns  molecule column size
// @param  int  max-columns  denominator column size
//
_row__col(columns = null, max-columns = null)

//
// Setting column width
//
// @param  length  width
//
_row__col--width(width)

//
// Setting column offset
//
// @param  int  columns  molecule offset size
// @param  int  max-columns  denominator offset size
//
_row__col--offset(columns = null, max-columns = null)

//
// Adding margin for the column
//
// @param  int  coefficient
//
_row__col--margin(coefficient)

//
// Adding margin for the row
//
// @param  int  coefficient
//
_row--margin(coefficient)

.c-row {
  _row();
  &__col {
    _row__col();
    &--1-3 {
      _row__col(1, 3);
    }
    &--offset-1-3 {
      _row__col--offset(1, 3);
    }
  }
}
